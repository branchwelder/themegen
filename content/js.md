var fibonacci = (function () {
  var cache = new Map();
  return function (n) {
    if (n == 0 || n == 1) {
      cache.set(n, n);
      return n;
    }
    var v1 = cache.get(n - 1) ? cache.get(n - 1) : fibonacci(n - 1);
    var v2 = cache.get(n - 2) ? cache.get(n - 2) : fibonacci(n - 2);
    var nVal = v1 + v2;
    cache.set(n, nVal);
    return nVal;
  };
})();
